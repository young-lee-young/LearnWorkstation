### 并发问题

* 问题

线程 M 去全局队列抓取协程 G，有并发问题

* 解决

使用本地队列，每次去全局队列拿一批 G 协程

* 队列抽象结构体（runtime/runtime2.go/p）

```go
package runtime

type p struct {
	id          int32
	status      uint32 // one of pidle/prunning/...
	link        puintptr
	schedtick   uint32     // incremented on every scheduler call
	syscalltick uint32     // incremented on every system call
	sysmontick  sysmontick // last tick observed by sysmon
	m           muintptr   // m 的 指针，指向提供服务的线程
	mcache      *mcache    // 本地内存索引，每个 mcache 为 136 种 mspan
	pcache      pageCache
	raceprocctx uintptr

	deferpool    [5][]*_defer 
	deferpoolbuf [5][32]*_defer

	goidcache    uint64
	goidcacheend uint64

	// Queue of runnable goroutines. Accessed without lock.
	runqhead uint32        // 队列头
	runqtail uint32        // 队列尾
	runq     [256]guintptr // 协程队列

	runnext guintptr // 下一个可用协程指针

	// Available G's (status == Gdead)
	gFree struct {
		gList
		n int32
	}

	sudogcache []*sudog
	sudogbuf   [128]*sudog

	// Cache of mspan objects from the heap.
	mspancache struct {
		len int
		buf [128]*mspan
	}

	tracebuf traceBufPtr

	traceSweep                 bool
	traceSwept, traceReclaimed uintptr
	palloc                     persistentAlloc // per-P to avoid mutex
	_                          uint32          // Alignment for atomic fields below
	timer0When                 uint64
	timerModifiedEarliest      uint64
	gcAssistTime               int64 // Nanoseconds in assistAlloc
	gcFractionalMarkTime       int64 // Nanoseconds in fractional mark worker (atomic)
	gcMarkWorkerMode           gcMarkWorkerMode
	gcMarkWorkerStartTime      int64
	gcw                        gcWork
	wbBuf                      wbBuf
	runSafePointFn             uint32 // if 1, run sched.safePointFn at next safe point
	statsSeq                   uint32
	timersLock                 mutex
	timers                     []*timer
	numTimers                  uint32
	adjustTimers               uint32
	deletedTimers              uint32
	timerRaceCtx               uintptr
	preempt                    bool // 是否被抢占
	pad                        cpu.CacheLinePad
}
```

* 协程选择过程（runtime/proc.go/schedule()）

```go
package runtime

func schedule() {
    var gp *g
    
    // 从本地队列获取协程
 	if gp == nil {
		gp, inheritTime = runqget(p)
	}
	
	// 本地队列未找到协程，从全局队列里找
	if gp == nil {
		gp, inheritTime = findrunnable() // blocks until work is available
	}
}

// 从本地队列获取
func runqget(p *p) {
	return p.runnext
}

func findrunnable() {
    // 再次尝试从本地队列获取
	runqget(p)
	
	// 从全局队列获取
	globrunqget()
	
	// 从其他协程偷协程
	stealWork()
}
```

* 作用

1. M 与 G 之间的中介
2. P 持有一些 G，使得每次获取 G 的时候不用从全局找
3. 减少了并发冲突的情况
