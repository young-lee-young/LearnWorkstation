

### 配置文件

```
load_module /usr/lib64/nginx/modules/ngx_stream_module.so;

stream{
	# 代理端口
    server{
        listen 8001;
        proxy_pass proxy_a;
    }

    # 向被代理机器转发
    upstream proxy_a{
        server 127.0.0.1:8570;
		server 127.0.0.1:8580;
        server 127.0.0.1:8590;
    }
}

events {
    worker_connections 1024;
}
```
