# 模型的保存和读取


### 保存

* 方式 1 代码示例

```python
import torch
import torchvision


vgg16 = torchvision.models.vgg16(weights=None)


# 方式 1
model_dir = "/Users/lee/Workstation/DeepLearning/model/vgg16/vgg16_model1.pth"
# 保存网络模型结构和参数
torch.save(vgg16, model_dir)


# 方式 2
model_dir = "/Users/lee/Workstation/DeepLearning/model/vgg16/vgg16_model2.pth"
# 不保存网络模型的结构，只保存网络模型的参数到 dict
# 官方推荐的保存方式
torch.save(vgg16.state_dict(), model_dir)
```


### 读取

* 代码示例

```python
import torch
import torchvision


model1_dir = "/Users/lee/Workstation/DeepLearning/model/vgg16/vgg16_model1.pth"
# 加载模型结构和参数
vgg16_model1 = torch.load(model1_dir, weights_only=False)
print(vgg16_model1)


model2_dir = "/Users/lee/Workstation/DeepLearning/model/vgg16/vgg16_model2.pth"
vgg16_model2 = torchvision.models.vgg16()
vgg16_model2.load_state_dict(torch.load(model2_dir))
print(vgg16_model2)
```
