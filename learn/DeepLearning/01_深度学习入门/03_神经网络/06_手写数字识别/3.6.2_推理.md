### 公共代码

```python
import numpy as np
import pickle

pickle_dir = "/learn/DeepLearning"


def init_network():
    with open(pickle_dir + "/01_深度学习入门/03_神经网络/06_手写数字识别/sample_weight.pkl", "rb") as f:
        network = pickle.load(f)
    return network


def predict(network, x):
    W1, W2, W3 = network["W1"], network["W2"], network["W3"]
    b1, b2, b3 = network["b1"], network["b2"], network["b3"]

    a1 = np.dot(x, W1) + b1
    z1 = sigmoid(a1)

    a2 = np.dot(z1, W2) + b2
    z2 = sigmoid(a2)

    a3 = np.dot(z2, W3) + b3
    z3 = softmax(a3)

    return z3
```


### 代码实现

```python
import numpy as np

network = init_network()

(train_img, train_label), (test_img, test_label) = load_mnist()

correct = 0

# 循环所有图片
for i in range(len(test_img)):
    y = predict(network, test_img[i])

    # 获取概率最高的元素索引
    p = np.argmax(y)
    
    # 判断推理的是否正确
    if p == test_label[i]:
        correct += 1

print("correct/all: {}/{}".format(correct, len(test_img)))
```
