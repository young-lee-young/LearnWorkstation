# 乘法层


### 代码实现

* 乘法层实现

```python
class MulLayer:
    def __init__(self):
        self.x = None
        self.y = None

    # 前向传播
    def forward(self, x, y):
        self.x = x
        self.y = y
        out = x * y
        return out

    # 反向传播
    def backward(self, dout):
        dx = dout * self.y
        dy = dout * self.x
        return dx, dy
```

* 乘法层应用

```python
apple_unit = 100  # 苹果单价
apple_num = 2  # 苹果数量
tax = 1.1  # 税率

mul_apple_layer = MulLayer()
mul_tax_layer = MulLayer()

# 前向传播
apple_price = mul_apple_layer.forward(apple_unit, apple_num)
price = mul_tax_layer.forward(apple_price, tax)
print(price)

# 反向传播
d_price = 1
d_apple_price, d_tax = mul_tax_layer.backward(d_price)
d_apple_unit, d_apple_num = mul_apple_layer.backward(d_apple_price)
print(d_apple_unit, d_apple_num, d_tax)
```
