# 加法层

### 代码实现

* 加法层实现

```python
class AddLayer:
    def __init__(self):
        pass

    # 前向传播
    def forward(self, x, y):
        return x + y

    # 反向传播
    def backward(self, dout):
        dx = dout * 1
        dy = dout * 1
        return dx, dy
```

* 加法层应用

```python
apple_unit = 100  # 苹果单价
apple_num = 2  # 苹果数量
orange_unit = 150  # 橘子单价
orange_num = 3  # 橘子数量
tax = 1.1  # 税率

# 初始化层
mul_apple_layer = MulLayer()
mul_orange_layer = MulLayer()
add_apple_orange_layer = AddLayer()
mul_tax_layer = MulLayer()

# 前向传播
apple_price = mul_apple_layer.forward(apple_unit, apple_num)
orange_price = mul_orange_layer.forward(orange_unit, orange_num)
apple_orange_price = add_apple_orange_layer.forward(apple_price, orange_price)
price = mul_tax_layer.forward(apple_orange_price, tax)
print(price)

# 反向传播
d_price = 1
d_add_apple_orange_price, d_tax = mul_tax_layer.backward(d_price)
d_apple_price, d_orange_price = add_apple_orange_layer.backward(d_add_apple_orange_price)
d_apple_unit, d_apple_num = mul_apple_layer.backward(d_apple_price)
d_orange_unit, d_orange_num = mul_orange_layer.backward(d_orange_price)
print(d_apple_unit, d_apple_num, d_orange_unit, d_orange_num, d_tax)
```
