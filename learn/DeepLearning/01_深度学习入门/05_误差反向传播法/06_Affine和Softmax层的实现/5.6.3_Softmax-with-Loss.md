# Softmax with Loss


### 代码实现

```python
import numpy as np


def softmax(a):
    c = np.max(a)
    exp = np.exp(a - c)
    exp_sum = np.sum(exp)
    y = exp / exp_sum
    return y


def cross_entropy_error(y, t):
    delta = 1e-7
    return -np.sum(t * np.log(y + delta))


class SoftmaxWithLoss:
    def __init__(self):
        self.loss = None
        self.y = None
        self.t = None

    def forward(self, x, t):
        self.t = t
        self.y = softmax(x)
        self.loss = cross_entropy_error(self.y, self.t)
        return self.loss

    def backward(self, dout=1):
        batch_size = self.t.shape[0]
        # 注意除以 batch size
        dx = (self.y - self.t) / batch_size
        return dx
```
