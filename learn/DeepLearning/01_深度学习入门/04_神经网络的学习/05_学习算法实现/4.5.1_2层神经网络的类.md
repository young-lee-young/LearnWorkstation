# 2 层神经网络的类

### 代码实现

```python
import numpy as np


def sigmoid(a):
    return 1 / (1 + np.exp(-a))


def softmax(a):
    c = np.max(a)
    exp = np.exp(a - c)
    exp_sum = np.sum(exp)
    y = exp / exp_sum
    return y


# 交叉熵误差
def cross_entropy_error(y, t):
    delta = 1e-7
    return -np.sum(t * np.log(y + delta))


# 计算梯度
def numerical_gradient(f, x):
    h = 1e-4
    # 生成一个和 x shape 相同的数组
    ret = np.zeros_like(x)

    it = np.nditer(x, flags=['multi_index'], op_flags=['readwrite'])
    while not it.finished:
        idx = it.multi_index

        tmp = x[idx]

        x[idx] = tmp + h
        fx1 = f(x)

        x[idx] = tmp - h
        fx2 = f(x)

        # 求数值微分
        ret[idx] = (fx1 - fx2) / (2 * h)

        # 恢复数据
        x[idx] = tmp

        it.iternext()

    return ret


class TwoLayerNet:
    # input_size：输入层神经元数量
    # hidden_size：隐藏层神经元数量
    # output_size：输出层神经元数量
    # weight_init_std: 用均值为 0，标准差为 0.01 的正态分布来初始化权重
    def __init__(self, input_size, hidden_size, out_size, weight_init_std=0.01):
        self.params = {}

        self.params["W1"] = weight_init_std * np.random.randn(input_size, hidden_size)
        self.params["b1"] = np.zeros(hidden_size)
        self.params["W2"] = weight_init_std * np.random.randn(hidden_size, out_size)
        self.params["b2"] = np.zeros(out_size)

    # 预测
    def predict(self, x):
        W1, W2 = self.params["W1"], self.params["W2"]
        b1, b2 = self.params["b1"], self.params["b2"]

        a1 = np.dot(x, W1) + b1
        z1 = sigmoid(a1)

        a2 = np.dot(z1, W2) + b2
        z2 = softmax(a2)

        return z2

    # 计算损失
    def loss(self, x, t):
        y = self.predict(x)
        # 交叉熵误差
        return cross_entropy_error(y, t)

    # 计算正确率
    def accuracy(self, x, t):
        y = self.predict(x)
        y = np.argmax(y, axis=1)
        t = np.argmax(t, axis=1)
        # 正确率
        accuracy = np.sum(y == t) / float(x.shape[0])
        return accuracy

    # 数值微分
    def numerical_gradient(self, x, t):
        loss_W = lambda W: self.loss(x, t)

        grads = {}
        grads["W1"] = numerical_gradient(loss_W, self.params["W1"])
        grads["b1"] = numerical_gradient(loss_W, self.params["b1"])
        grads["W2"] = numerical_gradient(loss_W, self.params["W2"])
        grads["b2"] = numerical_gradient(loss_W, self.params["b2"])

        return grads

    def gradient(self, x, t):
        pass


net = TwoLayerNet(input_size=784, hidden_size=100, out_size=10)
x = np.random.rand(100, 784)
y = net.predict(x)

t = np.random.rand(100, 10)

grads = net.numerical_gradient(x, t)
print(grads["W1"].shape)
print(grads["b1"].shape)
print(grads["W2"].shape)
print(grads["b2"].shape)
```
