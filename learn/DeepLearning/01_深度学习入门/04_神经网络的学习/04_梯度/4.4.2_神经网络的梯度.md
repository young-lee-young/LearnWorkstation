import numpy as np


# 计算概率
def softmax(a):
    c = np.max(a)
    exp = np.exp(a - c)
    exp_sum = np.sum(exp)
    y = exp / exp_sum
    return y


# 计算损失
def cross_entropy_error(y, t):
    delta = 1e-7
    return -np.sum(t * np.log(y + delta))


# 计算梯度
def numerical_gradient(f, x):
    h = 1e-4
    # 生成一个和 x shape 相同的数组
    ret = np.zeros_like(x)

    it = np.nditer(x, flags=['multi_index'], op_flags=['readwrite'])
    while not it.finished:
        idx = it.multi_index

        tmp = x[idx]

        x[idx] = tmp + h
        fx1 = f(x)

        x[idx] = tmp - h
        fx2 = f(x)

        # 求数值微分
        ret[idx] = (fx1 - fx2) / (2 * h)

        # 恢复数据
        x[idx] = tmp

        it.iternext()

    return ret


class SimpleNet:
    def __init__(self):

        self.W = np.random.randn(2, 3)
