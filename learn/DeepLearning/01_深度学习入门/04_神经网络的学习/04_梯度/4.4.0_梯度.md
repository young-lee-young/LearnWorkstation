# 梯度


### 求梯度

```python
import numpy as np


def f(x):
    return np.sum(x ** 2)


# 数值梯度
def numerical_gradient(f, x):
    h = 1e-4
    # 生成一个和 x shape 相同的数组
    ret = np.zeros_like(x)

    # x.size 是 2
    for idx in range(x.size):
        tmp = x[idx]

        x[idx] = tmp + h
        fx1 = f(x)

        x[idx] = tmp - h
        fx2 = f(x)

        # 求数值微分
        ret[idx] = (fx1 - fx2) / (2 * h)

        # 恢复数据
        x[idx] = tmp

    return ret


# 求对 3.0 处的偏导时，4 为常数；求对 4.0 处的偏导时，3 为常数
x = np.array([3.0, 4.0])
ret = numerical_gradient(f, x)
# 输出：[6., 8.]
print(ret)
```
