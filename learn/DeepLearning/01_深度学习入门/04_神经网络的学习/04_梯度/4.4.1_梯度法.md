# 梯度法


### 公式

$$
x_0 = x_0 - \eta \frac{\partial f}{\partial x_0}
$$

$$
x_1 = x_1 - \eta \frac{\partial f}{\partial x_1}
$$

η 是学习率，是超参数，大小要合适


### 代码实现

```python
import numpy as np


def f(x):
    return np.sum(x ** 2)


def numerical_gradient(f, x):
    h = 1e-4
    # 生成一个和 x shape 相同的数组
    ret = np.zeros_like(x)

    # x.size 是 2
    for idx in range(x.size):
        tmp = x[idx]

        x[idx] = tmp + h
        fx1 = f(x)

        x[idx] = tmp - h
        fx2 = f(x)

        # 求数值微分
        ret[idx] = (fx1 - fx2) / (2 * h)

        # 恢复数据
        x[idx] = tmp

    return ret


def gradient_descent(f, x, lr=0.01, step=100):
    for i in range(step):
        # 计算器数值梯度
        grad = numerical_gradient(f, x)

        # 计算下一个点
        x -= lr * grad

    return x


# 求 f(x0 + x1) = x0 ** 2 + x1 ** 2，x0 和 x1 分别为多少，f 有最小值
x = np.array([-3.0, 4.0])
ret = gradient_descent(f, x, lr=0.1, step=100)
# 输出：[-6.11110793e-10  8.14814391e-10]
print(ret)
```
